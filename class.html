Structure with constructor, getters, setters and methods
class Person {
	// Constructor
	constructor (name, age) {
		this._name = name;
		this._age = age;
	}

	// Getters / Setters
	get name = () => { return this._name }
	get age = () => { return this._age }
	set name = (name) => { this._name = name }
	set age = (age) => { this._age = age }

	// Methods
	print = () => { console.log(`My name is ${person.name} and my age is ${person.age}`); }
	clear = () => { person = { name: '', age: 0 } }
}
